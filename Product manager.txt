function buildCollectionSelect(){
  const sel = $('f-coll-sel'); if (!sel) return;
  const opts = ['<option value="">— Select collection —</option>']
    .concat(BRANDS.map(b => `<option value="${esc(BRAND_TO_COLL[b])}">${esc(BRAND_TO_COLL[b])}</option>`));

function promptNewCollection(){
  const label = window.prompt('Collection name — one word only (e.g. "mini")');
  if (!label) return;
  const slug = slugify(label).replace(/-/g,'');
  if (!slug || !/^[a-z0-9]+$/.test(slug)) { toast('Use one word, letters/numbers only','err'); return; }
  if (CUSTOM_COLLECTIONS.some(c=>c.slug===slug) || BRANDS.some(b=>BRAND_TO_COLL[b]===slug)) {
    toast('Collection already exists','info');
  } else {
    CUSTOM_COLLECTIONS.push({label: slug, slug});
    buildCollectionSelect();
    toast(`Added collection: ${slug}`,'ok');
  }
  STATE.collSlug = slug;
  const sel = $('f-coll-sel'); if (sel) sel.value = slug;
}

function handleTagKey(e, field){
  if (e.key === 'Enter' || e.key === ',') {
    e.preventDefault();
    const val = e.target.value.replace(/,/g,'').trim().toUpperCase();
    if (val) { STATE[field].push(val); e.target.value=''; renderTags(field); autoSeo(); }
  } else if (e.key === 'Backspace' && !e.target.value && STATE[field].length) {
    STATE[field].pop(); renderTags(field);
  }
}
function renderTags(field){
  const host = $(field+'-tags'); if (!host) return;
  host.innerHTML = STATE[field].map((t,i) => `<span class="tag">${esc(t)}<button onclick="removeTag('${field}',${i})" aria-label="Remove">×</button></span>`).join('');
}
function removeTag(field, i){ STATE[field].splice(i,1); renderTags(field); autoSeo(); }

// ─────────────────────── SEO ───────────────────────
function autoSeo(){
  const brand = STATE.brand || gv('f-brand');
  const mpn = gv('f-mpn');
  if (!brand || !mpn) { v('f-seo-title',''); v('f-seo-desc',''); charCount('f-seo-title','cc-title',70); charCount('f-seo-desc','cc-desc',150); updateSeoPreview(); return; }
  const drive = STATE.drive;
  const SEO_PART_PHRASE = {'EPS Rack':'Power Steering Gear (EPS) Rack','Steering Rack':'Power Steering Rack','Steering Column':'Power Steering Column'};
  const phrase = SEO_PART_PHRASE[STATE.partType] || 'Power Steering Rack';
  const title = `${brand} OEM ${mpn} ${phrase}${drive==='LHD'?' LHD':''}`.slice(0,70);
  const models = STATE.compatRows.filter(r=>r.model).map(r=>r.model.trim()).filter((v,i,a)=>a.indexOf(v)===i).join(' ');
  const years = STATE.compatRows.map(r=>r.years||'').filter(Boolean);
  const yearFrom = years.length ? years.reduce((a,b) => { const ay=a.match(/(\d{4})/)?.[1]||''; const by=b.match(/(\d{4})/)?.[1]||''; return ay<by?a:b; }).match(/(\d{4})/)?.[1]||'' : '';
  const driveQual = drive==='LHD' ? 'left-hand drive ' : '';
  const short = {'EPS Rack':'EPS Rack','Steering Rack':'Steering Rack','Steering Column':'Steering Column'}[STATE.partType] || 'EPS Rack';
  const refs = STATE.oerefs.filter(r => r.toUpperCase() !== mpn.toUpperCase()).slice(0,4).join(', ');
  let desc = `Genuine ${brand} part ${mpn} OEM ${short}`;
  if (refs) desc += ` Cross-ref: ${refs}`;
  if (models && yearFrom) desc += ` fits ${driveQual}${models} models from ${yearFrom} onwards.`;
  v('f-seo-title', title); v('f-seo-desc', desc.slice(0,150));
  charCount('f-seo-title','cc-title',70); charCount('f-seo-desc','cc-desc',150);
  updateSeoPreview();
}
function updateSeoPreview(){
  $('seo-pt').textContent = gv('f-seo-title') || '—';
  $('seo-pd').textContent = gv('f-seo-desc') || '—';
}
function charCount(inputId, countId, limit){
  const len = ($(inputId)?.value || '').length;
  const el = $(countId); if (!el) return;
  el.textContent = `${len} / ${limit}`;
  el.className = 'char-count' + (len > limit ? ' over' : len > limit*0.9 ? ' warn' : '');
}

function slugify(s){
  return String(s||'').toLowerCase().replace(/[^a-z0-9]+/g,'-').replace(/^-+|-+$/g,'');
}
function generateProductHtml(){
  const brand = STATE.brand || gv('f-brand') || 'Brand';
  const collSlug = STATE.collSlug || gv('f-coll') || slugify(brand);
  const collLabel = brand;
  const name = gv('f-name') || 'Untitled product';
  const seoTitle = gv('f-seo-title') || name;
  const mpn = gv('f-mpn') || '—';
  const isNew = STATE.condition === 'new';
  const condLabel = isNew ? 'New' : 'Used';
  const drive = STATE.drive || 'RHD';
  const contactUrl = WHATSAPP_URL;
  const warrantyMonths = Number(STATE.warrantyMonths || (isNew ? DEFAULT_WARRANTY_NEW : DEFAULT_WARRANTY_USED));
  const returnsDays = Number(STATE.returnsDays || 14);
  const badgeTax     = STATE.badgeTax || 'TAX inc.';
  const badgeQuality = STATE.badgeQuality || (isNew ? DEFAULT_QUALITY_NEW : DEFAULT_QUALITY_USED);
  const cta1 = STATE.cta1 || 'UK REG/VIN/CHASSIS Fitment';
  const cta2 = STATE.cta2 || 'Coding & Support';
  const cta3 = STATE.cta3 || (isNew ? DEFAULT_CTA3_NEW : DEFAULT_CTA3_USED);
  const footerNote = STATE.footerNote || (isNew ? DEFAULT_FOOTER_NEW : DEFAULT_FOOTER_USED);
  const descNote = gv('f-desc-note') || (isNew ? DEFAULT_DESC_NEW : DEFAULT_DESC_USED);
  const intros = descNote.split(/\n{2,}|\r\n\r\n/).map(s=>s.trim()).filter(Boolean);
  const introHtml = intros.length
    ? intros.map(p => `      <p class="intro">${esc(p)}</p>`).join('\n')
    : `      <p class="intro">${esc(descNote)}</p>`;

  // images: hero = featured (or first), thumbs = rest
  const imgs = STATE.images.slice();
  const featuredIdx = Math.max(0, imgs.findIndex(i => i.featured));
  const heroImg = imgs[featuredIdx] || imgs[0] || null;
  const thumbImgs = imgs.filter((_, i) => i !== (heroImg ? imgs.indexOf(heroImg) : -1));
  const galleryHtml = `
  <div class="gallery">

    ${thumbImgs.length

    : '';
  const skipHtml = '';

  const compatRowsHtml = STATE.compatRows
    .filter(r => r.make || r.model)
    .map(r => `        <tr>
          <td>${esc(r.make||'')}</td>
          <td>${esc(r.model||'')}</td>
          <td>${esc(r.platform||'')}</td>
          <td>${esc(r.years||'')}</td>
          <td>${esc(r.drive||drive)}</td>
        </tr>`).join('\n');

  const compatBlock = compatRowsHtml ? `
  <div class="rule mt-16"></div>

  <div class="s-hd">
    <h3>Vehicle compatibility</h3>
  </div>

  <div class="t-wrap">
    <table class="ct">
      <thead>
        <tr>
          <th>Make</th>
          <th>Model</th>
          <th>Chassis</th>
          <th>Years</th>
          <th>Drive</th>
        </tr>
      </thead>
      <tbody>
${compatRowsHtml}
      </tbody>
    </table>
  </div>

  <p class="note">Can't find your model? <a href="${esc(contactUrl)}">Send VIN via WhatsApp</a> — we'll confirm fast.</p>
` : '';

  const deliveryRowsHtml = STATE.deliveryRows
    .filter(r => r.zone || r.rate)
    .map(r => `        <tr><td>${esc(r.zone||'')}</td><td>${esc(r.rate||'')}</td><td>${esc(r.handling||'')}</td><td>${esc(r.transit||'')}</td></tr>`)
    .join('\n');

  const html = `<div class="pg">
  <nav class="bc" aria-label="Breadcrumb">
    <a href="https://powersteeringrackdirect.com">PSRD OEM Steering Racks</a><span class="bc-s">›</span>

  <div class="badges">

    <span class="bdg b-drv${drive==='LHD'?' is-lhd':''}">${esc(STATE.badgeDrive || drive)}</span>
    <span class="bdg b-war">${esc(warrantyMonths)}-M Warranty</span>
  </div>

  <div class="rule"></div>

    <h2>Specification</h2>
  </div>

  <div class="ctas">
    <a href="${esc(contactUrl)}">${esc(cta1)}</a>
    <a href="${esc(contactUrl)}">${esc(cta2)}</a>
    <a class="primary" href="${esc(contactUrl)}">${esc(cta3)}</a>
  </div>
${compatBlock}
  <div class="rule mt-16"></div>

  <div class="s-hd">
    <h3>Delivery &amp; returns</h3>
  </div>

  <div class="t-wrap">
    <table class="ct">
      <thead>
        <tr>
          <th>Zone</th>
          <th>Standard rate</th>

      </thead>
      <tbody>
${deliveryRowsHtml || '        <tr><td colspan="4" style="text-align:center;color:var(--pg-ink-4)">No delivery zones configured</td></tr>'}
      </tbody>
    </table>
  </div>

  <p class="note">Buyer is responsible for all customs fees, import duties &amp; taxes.</p>

  <p class="note">Warranty: ${warrantyMonths} months from date of purchase. Covers manufacturing defects, non-working or damaged parts. Does not cover mechanical damage, incorrect installation, or misuse.</p>
  <p class="note">${esc(footerNote)}</p>

  return html;
}

let _previewMode = 'rendered';
function setPreviewMode(mode){
  _previewMode = mode;
  document.querySelectorAll('#preview-mode-seg button').forEach(b=>{
    b.classList.toggle('is-on', b.dataset.v === mode);
  });
  $('preview-frame').style.display  = mode === 'rendered' ? 'block' : 'none';
  $('preview-content').style.display = mode === 'rendered' ? 'none' : 'block';
}
function previewProduct(){
  const html = generateProductHtml();
  const fullDoc = generateFullPreviewDoc();
  $('preview-content').textContent = html;
  const iframe = $('preview-frame');
  iframe.srcdoc = fullDoc;
  setPreviewMode('rendered');
  openModal('preview-modal');
}
function openPreviewInNewTab(){
  const fullDoc = generateFullPreviewDoc();
  const blob = new Blob([fullDoc], {type:'text/html'});
  const url = URL.createObjectURL(blob);
  window.open(url, '_blank');
  setTimeout(()=>URL.revokeObjectURL(url), 30000);
}
function copyHtml(){
  const html = generateProductHtml();
  if (navigator.clipboard && navigator.clipboard.writeText) {
    navigator.clipboard.writeText(html).then(
      () => toast('HTML copied to clipboard', 'ok'),
      () => fallbackCopy(html)
    );
  } else { fallbackCopy(html); }
}
function fallbackCopy(text){
  const ta = document.createElement('textarea');
  ta.value = text; ta.style.position='fixed'; ta.style.opacity='0';
  document.body.appendChild(ta); ta.select();
  try { document.execCommand('copy'); toast('HTML copied','ok'); }
  catch(e){ toast('Copy failed — select & copy manually','err'); }
  document.body.removeChild(ta);
}
function publishProduct(){
  openModal('publish-modal');
  $('pub-title').textContent = 'Publishing…';
  $('pub-sub').textContent = 'Please wait — this can take ~5 seconds.';
  $('pub-close-btn').style.display = 'none';
  [1,2,3,4].forEach(n => {
    const d=$('step-dot-'+n), l=$('step-lbl-'+n); d.className='step-dot'; l.className='step-lbl';
  });
  let n = 0;
  const tick = () => {
    n++;
    if (n>1) { $('step-dot-'+(n-1)).className='step-dot done'; $('step-dot-'+(n-1)).textContent='✓'; $('step-lbl-'+(n-1)).className='step-lbl done'; }
    if (n>4) {
      $('pub-title').textContent = 'Published';
      $('pub-sub').textContent = 'Your product is live.';
      $('pub-close-btn').style.display = 'flex';
      return;
    }
    $('step-dot-'+n).className='step-dot active';
    $('step-lbl-'+n).className='step-lbl active';
    setTimeout(tick, 900);
  };
  setTimeout(tick, 300);
}
